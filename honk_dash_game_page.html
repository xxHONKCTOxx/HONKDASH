Press SPACE to flap the goose and navigate through gaps. Avoid hitting the baseball bats or the screen edges. Score points by passing obstacles. In the Game Over screen, press SPACE to submit a high score or ENTER to restart.   function preload() {

  function setup() {
    let canvas = createCanvas(400, 600);
    canvas.parent('gameCanvas');
    goose = new Goose();
    obstacles.push(new Obstacle());
  }

  function draw() {
    if (gameState === 'start') {
      background(135, 206, 235);
      textSize(32);
      textAlign(CENTER);
      fill(0);
      text('Honk Dash', width / 2, height / 2 - 20);
      textSize(20);
      text('Press SPACE to start', width / 2, height / 2 + 20);
      displayHighScores(width - 150, 50);
    } else if (gameState === 'playing') {
      background(135, 206, 235);
      goose.update();
      goose.show();
      
      if (frameCount % 103 === 0) {
        obstacles.push(new Obstacle());
      }
      
      let collision = false;
      for (let ob of obstacles) {
        ob.update();
        ob.show();
        if (ob.hits(goose)) {
          collision = true;
        }
      }
      
      if (collision || goose.y <= 0 || goose.y >= height - goose.size) {
        canSubmitHighScore = isHighScore(score);
        gameState = 'gameover';
      } else {
        for (let i = obstacles.length - 1; i >= 0; i--) {
          let ob = obstacles[i];
          if (goose.x > ob.x + ob.width && !ob.passed) {
            ob.passed = true;
            obstaclesPassed++;
            score += 1;
          }
          if (ob.x < -ob.width) {
            obstacles.splice(i, 1);
          }
        }
      }
      
      textSize(24);
      fill(0);
      textAlign(LEFT);
      text('Score: ' + score, 10, 30);
    } else if (gameState === 'enterName') {
      background(135, 206, 235);
      if (!nameInput) {
        nameInput = createInput();
        nameInput.position(width / 2 - 50, height / 2);
        submitButton = createButton('Submit');
        submitButton.position(width / 2 - 25, height / 2 + 30);
        submitButton.mousePressed(submitName);
      }
      textSize(24);
      textAlign(CENTER);
      fill(0);
      text('New High Score!', width / 2, height / 2 - 40);
      text('Enter your name:', width / 2, height / 2 - 10);
      text('Score: ' + score, width / 2, height / 2 + 60);
    } else if (gameState === 'gameover') {
      background(135, 206, 235);
      textSize(32);
      textAlign(CENTER);
      fill(0);
      text('Game Over!', width / 2, height / 2 - 20);
      text('Score: ' + score, width / 2, height / 2 + 20);
      textSize(20);
      text('Press ENTER to restart', width / 2, height / 2 + 60);
      displayHighScores(width - 150, 50);
      if (canSubmitHighScore) {
        textSize(16);
        text('New High Score! Press SPACE to submit your name or ENTER to play again.', width / 2, height / 2 + 90);
      }
    }
  }

  function keyPressed() {
    if (key === ' ' && gameState === 'start') {
      gameState = 'playing';
    } else if (key === ' ' && gameState === 'playing') {
      goose.flap();
    } else if (keyCode === ENTER && gameState === 'gameover') {
      canSubmitHighScore = false;
      goose = new Goose();
      obstacles = [];
      score = 0;
      obstaclesPassed = 0;
      gameState = 'start';
    } else if (key === ' ' && gameState === 'gameover' && canSubmitHighScore) {
      gameState = 'enterName';
    }
  }

  class Goose {
    constructor() {
      this.y = height / 2;
      this.x = 100;
      this.velocity = 0;
      this.gravity = 0.6;
      this.lift = -8;
      this.size = 30;
      this.wingAngle = 0;
    }
    
    update() {
      this.velocity += this.gravity;
      this.y += this.velocity;
      
      if (this.y > height - this.size) {
        this.y = height - this.size;
        this.velocity = 0;
      }
      if (this.y < 0) {
        this.y = 0;
        this.velocity = 0;
      }
      
      if (this.wingAngle > 0) {
        this.wingAngle -= 0.1;
        if (this.wingAngle < 0) this.wingAngle = 0;
      }
    }
    
    show() {
      fill(255);
      ellipse(this.x, this.y, this.size, this.size * 0.8);
      
      push();
      translate(this.x - 15, this.y);
      rotate(-this.wingAngle);
      ellipse(10, 0, 20, 10);
      pop();
      
      push();
      translate(this.x + 15, this.y);
      rotate(this.wingAngle);
      ellipse(-10, 0, 20, 10);
      pop();
      
      stroke(255);
      strokeWeight(4);
      line(this.x, this.y - 10, this.x, this.y - 25);
      noStroke();
      strokeWeight(1);
      
      fill(255);
      ellipse(this.x, this.y - 25, 15, 15);
      
      fill(0);
      ellipse(this.x + 2, this.y - 27, 4, 4);
      
      fill(255, 165, 0);
      triangle(this.x + 5, this.y - 25, this.x + 15, this.y - 23, this.x + 5, this.y - 21);
    }
    
    flap() {
      this.velocity = this.lift;
      this.wingAngle = PI / 4;
    }
  }

  class Obstacle {
    constructor() {
      this.width = 60;
      this.x = width;
      let baseSpacing = 156;
      let baseSpeed = 1.5;
      let difficultyLevel = Math.floor(obstaclesPassed / 10);
      this.spacing = Math.max(100, baseSpacing - difficultyLevel * 10);
      this.speed = baseSpeed + difficultyLevel * 0.3;
      this.top = random(50, height - this.spacing - 50);
      this.passed = false;
    }
    
    update() {
      this.x -= this.speed;
    }
    
    show() {
      for (let y = 15; y < this.top; y += 60) {
        drawBat(this.x + 15, y);
        drawBat(this.x + 45, y);
      }
      for (let y = height - 15; y > this.top + this.spacing; y -= 60) {
        drawBat(this.x + 15, y);
        drawBat(this.x + 45, y);
      }
    }
    
    hits(goose) {
      for (let y = 15; y < this.top; y += 60) {
        if (rectCircleCollide(this.x + 15 - 5, y - 20, 10, 40, goose.x, goose.y, goose.size / 2) ||
            rectCircleCollide(this.x + 45 - 5, y - 20, 10, 40, goose.x, goose.y, goose.size / 2)) {
          return true;
        }
      }
      for (let y = height - 15; y > this.top + this.spacing; y -= 60) {
        if (rectCircleCollide(this.x + 15 - 5, y - 20, 10, 40, goose.x, goose.y, goose.size / 2) ||
            rectCircleCollide(this.x + 45 - 5, y - 20, 10, 40, goose.x, goose.y, goose.size / 2)) {
          return true;
        }
      }
      return false;
    }
  }

  function drawBat(x, y) {
    fill(139, 69, 19); // Brown for bat body
    rect(x - 5, y - 20, 10, 30); // Bat body
    fill(169, 169, 169); // Gray for handle
    rect(x - 3, y + 10, 6, 10); // Handle
  }

  function rectCircleCollide(rectX, rectY, rectWidth, rectHeight, circleX, circleY, circleRadius) {
    let closestX = Math.max(rectX, Math.min(circleX, rectX + rectWidth));
    let closestY = Math.max(rectY, Math.min(circleY, rectY + rectHeight));
    let distanceX = circleX - closestX;
    let distanceY = circleY - closestY;
    return (distanceX * distanceX + distanceY * distanceY) < (circleRadius * circleRadius);
  }

  function loadHighScores() {
    let scores = localStorage.getItem('highScores');
    if (scores) {
      try {
        return JSON.parse(scores);
      } catch (e) {
        console.error('Error parsing high scores:', e);
        return [];
      }
    }
    return [];
  }

  function saveHighScores(scores) {
    try {
      localStorage.setItem('highScores', JSON.stringify(scores));
    } catch (e) {
      console.error('Error saving high scores:', e);
    }
  }

  function isHighScore(score) {
    let highScores = loadHighScores();
    if (highScores.length === 0 || highScores.length < 5) {
      return true;
    }
    let minScore = Math.min(...highScores.map(s => s.score));
    return score >= minScore;
  }

  function submitName() {
    let name = nameInput.value().trim();
    if (name) {
      let highScores = loadHighScores();
      highScores.push({name: name, score: score});
      highScores.sort((a, b) => b.score - a.score);
      if (highScores.length > 5) {
        highScores = highScores.slice(0, 5);
      }
      saveHighScores(highScores);
      nameInput.remove();
      submitButton.remove();
      nameInput = null;
      submitButton = null;
      canSubmitHighScore = false;
      goose = new Goose();
      obstacles = [];
      score = 0;
      obstaclesPassed = 0;
      gameState = 'start';
    }
  }

  function displayHighScores(x, y) {
    let highScores = loadHighScores();
    textSize(16);
    textAlign(LEFT);
    fill(0);
    text('High Scores:', x, y);
    for (let i = 0; i < highScores.length; i++) {
      text(`${i + 1}. ${highScores[i].name}: ${highScores[i].score}`, x, y + 20 * (i + 1));
    }
  }
